data {
	int<lower=0> J; //kysymysten maara
        int n[J]; //vastausten maara jokaiselle kysymykselle
        int y[J]; //oikeiden vastausten maara jokaiselle kysymykselle
}
parameters {
        real<lower=0> alpha;
        real<lower=0> beta;
        vector[J] theta; //todennakoisyys etta vastaus on oikein jokaiselle kysymykselle
}
model {
        alpha ~ normal(5, 1);
        beta ~ normal(5, 1);
        
        //priors
        for (j in 1:J) {
            theta[j] ~ beta(alpha, beta);
        }
        
        //likelihood
        for (j in 1:J) {
            y[j] ~ binomial(n[j], theta[j]);
        }
}
